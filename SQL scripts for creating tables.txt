//light

CREATE TABLE `licenta`.`lights`(
    `id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `light` FLOAT(10) NOT NULL,
    `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
)

//temp and humidity air

CREATE TABLE `licenta`.`temps`(
    `id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `temperature` FLOAT(10) NOT NULL,
	`humidity` FLOAT(10) NOT NULL,
  `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
)

//soil moisture 

CREATE TABLE `licenta`.`soils`(
    `id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `moist` FLOAT(10) NOT NULL,
  `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
)
